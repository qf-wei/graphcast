# 1.  Throw away every GPU lib you installed via conda
conda remove --force cuda* nvidia* cudnn --yes

# 2.  Make sure no stale *.so is left in $CONDA_PREFIX/lib
rm -f $CONDA_PREFIX/lib/libcudnn*        \
      $CONDA_PREFIX/lib/libcusparse*    \
      $CONDA_PREFIX/lib/libnvJitLink*

# 3.  Pull the JAX split-plugin wheels **AND ONLY their matching nvidia-* wheels**
pip install --no-cache-dir --force-reinstall \
  "jax==0.6.2" \
  --extra-index-url https://storage.googleapis.com/jax-releases/jax_cuda_releases.html \
  jaxlib==0.6.2 \
  jax-cuda12-plugin==0.6.2 \
  nvidia-cublas-cu12==12.9.1.4 \
  nvidia-cusparse-cu12==12.5.10.65 \
  nvidia-cudnn-cu12==9.10.2.21 \
  nvidia-nvjitlink-cu12==12.9.86 \
  nvidia-cuda-runtime-cu12==12.9.79

# 4.  ONE single LD_LIBRARY_PATH pointing *only* at those wheels
export LD_LIBRARY_PATH=$CONDA_PREFIX/lib/python3.12/site-packages/nvidia:$CONDA_PREFIX/lib

# 5.  Test
python - <<'PY'
import jax
from jax.extend import backend
print("Backend :", backend.get_backend().platform)
print("Devices :", jax.devices())
PY
